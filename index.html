# ชื่อโครงการ (คลิกบรรทัดเพื่อดูรายละเอียด) ยุทธศาสตร์ งบอนุมัติ (บาท) ใช้ไป (บาท) PO (บาท) ใช้+PO (บาท) คงเหลือสุทธิ สถานะ ตัวชี้วัด ผู้แก้ไขล่าสุด จัดการ
เชื่อมต่อ Google Apps Script (GAS)
วาง URL ของ Google Apps Script Web App ที่นี่
(ได้จาก Apps Script → Deploy → New deployment → Web app)
https://script.google.com/macros/s/[DEPLOYMENT_ID]/exec
สถานะ: ยังไม่ได้ตั้งค่า GAS URL
Sync ข้อมูลกับ Google Sheet
ส่งข้อมูลโครงการทั้งหมดที่มีอยู่ขึ้น Google Sheet ทันที (เขียนทับข้อมูลเดิม)
ส่งออกข้อมูล (CSV)
ดาวน์โหลดข้อมูลโครงการทั้งหมดเป็น CSV แล้วนำเข้า Google Sheets ได้เลย
รูปแบบข้อมูลที่รองรับ
Sheet ควรมีหัวคอลัมน์แถวแรกดังนี้:

id | name | strategy | budget | spent | po | status | result | kpi | problems | solutions | quarter | owner | committees | image_count | images_json

• strategy = 1, 2, 3, 4 หรือ 5 (งบบริหารสำนักงาน)
• budget, spent, po = จำนวนเงิน (ตัวเลข ไม่ใส่ comma)
• status = done / progress / pending
• kpi = ตัวชี้วัด คั่นด้วย | (pipe)
• quarter = 1, 2, 3, 4 หรือ all
• owner = ผู้รับผิดชอบ / หน่วยงาน
• committees = JSON array เช่น ["นโยบายและแผนงาน","วิชาการ"]
• images_json = JSON array ของรูปภาพ (base64)
📸 วิธีการส่งรูปภาพขึ้น Google Sheet
รูปภาพจะถูกแปลงเป็น Base64 และส่งไปพร้อมข้อมูลโครงการผ่าน GAS
Google Apps Script ของคุณต้องรองรับ field images ใน payload

ตัวอย่าง GAS Code ที่รองรับรูปภาพ:

function doPost(e) { const payload = JSON.parse(e.postData.contents); const sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName('Projects') || SpreadsheetApp.getActiveSpreadsheet() .getSheets()[0]; if (payload.action === 'save' || payload.action === 'update') { const d = payload.data; const images = d.images || []; // บันทึกจำนวนรูปและ JSON ลง Sheet const imgCount = images.length; const imgJson = JSON.stringify(images.map(i=>({name:i.name,dataUrl:i.dataUrl}))); // หาแถวที่มี id ตรงกัน หรือเพิ่มแถวใหม่ const data = sheet.getDataRange().getValues(); let rowIdx = -1; for(let i=1;i0) sheet.getRange(rowIdx,1,1,row.length).setValues([row]); else sheet.appendRow(row); return ContentService.createTextOutput( JSON.stringify({success:true})).setMimeType(ContentService.MimeType.JSON); } // ... (action อื่นๆ เช่น getAll, delete, bulkSave) }
รีเซ็ตข้อมูลกลับค่าเริ่มต้น
ล้างข้อมูลที่บันทึกใน Browser และโหลดข้อมูลโครงการเริ่มต้นใหม่
⚠️ การกระทำนี้จะลบการแก้ไขทั้งหมด
💾 การบันทึกอัตโนมัติ
ข้อมูลโครงการจะถูกบันทึกลงใน Browser อัตโนมัติทุกครั้งที่มีการเพิ่ม แก้ไข หรือลบโครงการ
เมื่อตั้งค่า GAS URL แล้ว การเปลี่ยนแปลงทุกครั้งจะ Sync ขึ้น Google Sheet โดยอัตโนมัติ
